    = 13a-8b+12a+9b

    = 15i+14b+13k-30i-45j+51k
    = 15i-30i+14b+13k+51k-45j
    = -15i+14b+64k-45j
3. (3-17x+35z) + (4x+23y-9)
    = 3-17x+35z+4x+23y-9
    = -17x+4x+35z+23y+3-9
    = -13x+35z+23y-6
4. (42n+35m+7) - (50m-20n+9)
    = 42n+35m+7-50m+20n-9
    = 42n+20n+35m-50m+7-9
