→ x = 49°

Nilai y

→ y + 42° = 180° - (3x - 42°)

→ y + 42° = 180° - (3(49°) - 42°)

→ y + 42° = 180° - (147° - 42°)

→ y + 42° = 180° - 105°

→ y + 42° = 75°

→ y = 75° - 42°

→ y = 33°

→ x + y

→ 49° + 33°

→ 82°
