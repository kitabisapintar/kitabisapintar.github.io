nomor 1

sin 120° = sin (180° - 60°)

             = sin 60°

sin 120° =  1/2√3

cos 120° = cos (180° - 60°)

              = -cos 60°

cos 120° =  -1/2

tan 120° = tan (180° - 60°)

              = -tan 60°

tan 120° = √3

nomor 2

sin 210° = sin (180° + 30°)

             = -sin 30°

sin 210° =  -1/2

cos 210° = cos (180° + 30°)

              = -cos 30°

sin 210° =  -1/2√3

tan 210° = tan (180° + 30°)

              = tan 30°

tan 210° = √3

nomor 3

sin 315° = sin (360° - 45°)

             = -sin 45°

sin 315° =  -1/2√2

cos 315° = cos (360° - 45°)

              = cos 45°

sin 315° =  1/2√2

tan 315° = tan (360° - 45°)

             = -tan 45°
