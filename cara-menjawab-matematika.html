= 9 × 6 + 32 = 54 + 32 = 86°F

= 9 × 4 + 32 = 36 + 32 = 68°F

K = C + 273 = 20 + 273 = 293 K

C = [tex]\sf \frac{5}{9}[/tex] × (F - 32) = [tex]\sf \frac{5}{9}[/tex] × (86 - 32)

= [tex]\sf \frac{5}{9}[/tex] × 54 = 5 × 6 = 30°C

R = [tex]\sf \frac{4}{9}[/tex] × (F - 32) = [tex]\sf \frac{4}{9}[/tex] × 54 = 4 × 6 = 24°R

= 9 × 10 + 32 = 90 + 32 = 122°F
